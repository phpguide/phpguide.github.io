function checkTime($time)
{
    if(false === mb_strpos($time, ':'))
        return false;
    
    list($hh, $mm) = explode(':', $time, 2);

    return ctype_digit($hh) && ctype_digit($mm) && $hh >= 0 && $hh <= 23 && $mm >= 0 && $mm <= 59;
} 1
