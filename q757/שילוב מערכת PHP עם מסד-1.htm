אני צריך לשלב מסד נתונים..
זה הקוד PHP הקיים שקשור למסד נתונים :

$name_db = mysql_num_rows (mysql_query("SELECT * FROM files")) + 1;
$name = $_FILES["file".$up]["name"];
$dir = "upload/";
$save = $dir.$name_db.".".$type;
mysql_query("INSERT INTO VALUES ('', '$name', '$save', '$type')");
move_uploaded_file($_FILES["file".$up]["tmp_name"], $save);
