אם אין כלום במערך, אז הדפס\render ...

// מצב רגיל
  $items = array("pencil", "box", "cup");
 
  echo View::render_each('items.row', $items, 'item', 'raw|no items..'); // ידפיס את הview items.row 3 פעמים.
 
  // המערך ריק
  $items = array();
 
  echo View::render_each('items.row', $items, 'item', 'raw|no items..'); // ידפיס "no items.."
 
  echo View::render_each('items.row', $items, 'item', 'items.empty'); // ידפיס את view items.empty
