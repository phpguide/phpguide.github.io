define('SPARKPRICE', 4);

$totalamount = $tireqty * TIREPRICE
               + $oilqty * OILPRICE
         + $sparkqty * SPARKPRICE;
echo "Subtotal: $".number_format($totalamount,2)."<br />";

$taxrate = 0.10; //local sales tax is 10%

echo "Total including tax: $".number_format($totalamount,2)."<br />";
