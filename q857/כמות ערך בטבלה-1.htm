FROM `table`
GROUP BY `packid`
HAVING COUNT(`packid`) = (
    SELECT COUNT(`packid`) AS `c`
    FROM `table`
    GROUP BY `packid`
    ORDER BY `c` DESC
    LIMIT 1
);
