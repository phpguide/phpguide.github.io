$("#portfolioimages > .image").each(function(){
            var datasrc = $(this).find("img").attr("data-src");
           
            objImage = new Image();
            objImage.onLoad=imagesLoaded($(this), datasrc);
            objImage.src=datasrc;

            function imagesLoaded(image, src)
            {
                image.find("img").attr("src",src);
            }
        });
