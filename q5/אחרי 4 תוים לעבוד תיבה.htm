  <input />
  <input />

<script>
jQuery(function(){
 
  var inputs = $("#inputs input");
  inputs.each(function(index, item){
    $(item).keyup(function(){  
     
      $(this).val($(this).val().substr(0,4));
      if($(this).val().length >=4 && index!=inputs.length) $(inputs[index+1]).focus();
    });
  });

 });
</script>
