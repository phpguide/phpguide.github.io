תראה איך זה עובד ב YII, ששם שאילתות נבנות עם QueryBuilder בצורה הבאה:

$builder = new QueryBuilder();
$builder
    ->select('id, username, profile')
    ->from('tbl_user u')
    ->join('tbl_profile p', 'u.id=p.user_id')
    ->where('id=:id', array(':id'=>$id))
    ->queryRow();
