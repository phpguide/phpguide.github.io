function folder($path)
  {
    if($handle = opendir($path))
    {
      while (false !== ($file = readdir($handle)))
      {
        if($file != "." && $file != "..")
        {
          echo "<li>$file</li><br />";
          if(is_dir($file))
          {
            if ($file != "." && $file != "..")
              $scandir = scandir($file);
              foreach($scandir as $value)
              {
                if($value !== "." & $value !== "..")
                {
                  echo "-- ",$value,"<br />";            
                }
              }
          }
        }
      }
      closedir($handle);
    }
  }
