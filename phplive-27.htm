    '?' => 'y',

    // Latin symbols
    'Â©' => '(c)',

    // Greek

    '?' => 'I', '?' => 'K', '?' => 'L', '?' => 'M', '?' => 'N', '?' => '3', '?' => 'O', '?' => 'P',

    $str = str_replace(array_keys($char_map), $char_map, $str);
  }
